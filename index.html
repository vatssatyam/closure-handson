  var counter = 0;

  function IncreaseCounter() {

  }

  return IncreaseCounter;
}

Q2."let count = 0; (function () { if (count === 0) { let count = 1; console.log(count); // What is logged? } console.log(count); // What is logged? })();"

  if (count === 0) {

})();
